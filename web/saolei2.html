      # JavaScript实现基本逻辑

      # 分析

      1. 代码复用

      • 由于扫雷这个游戏，是由三个模式组成，分别为初级、中级和高级。如果一个个去写它们每个的代码，会显得代码臃肿庞大而且浪费时间。

      • 或者说你会写完某一个具体的代码逻辑，然后copy后稍作修改？ 那不如使用JavaScript 类！

      • JavaScript 类的使用方法十分简单，推荐阅读阮一峰的网络日志 (opens new window)了解一下JavaScript类的基本概念。

      1. 数据存储
      • 在扫雷这个游戏里，操作区域是由一个 n × n 组成的 n^2^ 个 格子，我个人感觉使用table表格比较合适。
      • 用JavaScript根据参数生成DOM以及计算格子应有的数据（是否为雷或者后面的数字是几）。
      • 需要一个布尔参数保存 格子是否被标记

      # 代码实现

      首先，还是要先在html文件最后（body标签之前）引入：<script src="game.js"></script> 。 *[引入]: 注意更改你自己的文件路径！

      # 创建类

      function Mine(tr, td, mineNum){
    this.tr = tr; // 行 
    this.td = td; // 列 
    this.mineNum = mineNum; // 雷的数量 

    this.squares = [];
    this.tds = []; // 所有单元格的DOM 
    this.surplusMine = mineNum; // 剩余雷的数量

    this.allRight = false; // 右击的小红旗是否全为红。，用来判断用户是否游戏成功。

    this.parent = document.querySelector('.games-area');
}

      定义了类Mine，有三个参数tr,td, mineNum；传入各自的行、列数和总的雷的数量。

      this.squares: 储存所有格子的信息，是一个二维数组。 this.parent: 获取容器（主区域）的DOM this.surplusMine: 剩余的雷的数量。

      # 创建DOM方法

      创建完类后，就需要给类添加方法，首先最重要的是要创建表格内的DOM，并把DOM插入页面中。 通过类的 prototype 可以给类 Mine 添加成员方法；

      Mine.prototype.createDom = function(){
    const table = document.createElement('table');

    for(let i=0;i<this.tr;i++){
        const domTr = document.createElement('tr');
        this.tds[i] = [];
        for(let j=0;j<this.td;j++){
            const domTd = document.createElement('td');
            this.tds[i][j] = domTd;
            domTr.appendChild(domTd);
        }
        table.appendChild(domTr);
    }
    this.parent.innerHTML = ''; // 清空，避免多次创建叠加
    this.parent.appendChild(table);
}

      在createDom方法中，使用双层循环创建行tr、列td的DOM，并插入进一个表格tableDOM，最后把tableDOM添加到主区域内。

      这样，只需要实例化类，然后调用createDom方法即可：

      var mine = new Mine(16,16,40);
mine.createDom();

      # 随机生成类的位置

      • 雷的位置需要随机生成，这时就需要生成一个随机数，这些随机数作为雷的位置。
      • 当然，这些随机数不是完全随意的，我们需要随机生成一个范围在行(this.tr)和列(this.td)范围之内的数。
      • 随机数也是一个二维数组。

      代码如下（方法不唯一）：

      Mine.prototype.randomNumber = function(){
    var square = new Array(this.tr*this.td); // 空数组
    for(let i=0;i<square.length;i++){
        square[i] = i;
    }
    square.sort(function(){return 0.5-Math.random()}); // 随机排序
    return square.slice(0,this.mineNum); // 返回其中 ‘雷的个数’ 个。
}

      写好生成雷的位置（随机数）的方法后，即可把雷的位置写进数组this.squares。 在init()方法内：

      Mine.prototype.init = function(){
	var rn = this.randomNumber(); // 随机生成雷的位置。

    // 下面循环用于把雷的位置写入 数组this.squares；
    // 注意：y——行 / x——列
    let n = 0;
    for(let i=0;i<this.tr;i++){
        this.squares[i] = [];
        for(let j=0;j<this.td;j++){
            n++;
            if(rn.indexOf(n) != -1){
                this.squares[i][j] = {type:'mine',x:j,y:i};
            }else{
                // value默认为0
                this.squares[i][j] = {type:'number',x:j,y:i,value:0}; 
            }
        }
    }
	this.createDom();
}
      • 其中，雷的数据为：{type:'mine',x:j,y:i}；
      • 而非雷的数据为：{type:'number',x:j,y:i,value:0}，有一个value参数

      # 计算雷周围的数字

      每个雷周围的八个（除了边界上的雷）格子，如果不是雷，就需要把默认数字0加上1；累加后的数字就是最终呈现的数字。 首先，需要写一个方法来确定每个雷的周围的数字格子的坐标。

      // getAround方法确定周围需要加数字的坐标。
// value默认为0，雷的周围（最多八个）数字加1，累加起来即为value的值。
Mine.prototype.getAround = function(square){
    let x = square.x;
    let y = square.y;
    /*
    x-1,y-1     x,y-1   x+1,y-1
    x-1,y       (x,y)    x+1,y
    x-1,y+1     x,y+1   x+1,y+1
    */
    let result = [];
    for(let i=x-1;i<=x+1;i++){
        for(let j=y-1;j<=y+1;j++){
            if(
                i<0 ||
                j<0 ||
                i>this.td-1 ||
                j>this.tr-1 ||
                (i==x&&j==y) ||
                this.squares[j][i].type == 'mine'
            ){
                continue;
            }
            result.push([j,i]); // not [j][i]!
        }
    }
    return result; // 返回雷周围的数字格子的坐标
}

      接着，需要写个方法，进行数字格子的value值的更新，即计算雷周围的数字。

      // 更新value —— 需要监控鼠标点击效果之后才去更新数据（防止用户在控制台提前看出！），所以要单独写个类的方法。
Mine.prototype.updateNumber = function(){
    for(let i=0;i<this.tr;i++){
        for(let j=0;j<this.td;j++){
            if(this.squares[i][j].type == 'number'){
                continue;
            }
            let loc = this.getAround(this.squares[i][j]);
            for(let item=0;item<loc.length;item++){
                let locx = loc[item][0];
                let locy = loc[item][1];
                this.squares[locx][locy].value += 1;
            }
        }
    }
}

      之后，在init()方法内调用updateNumber方法:

      this.updateNumber();

      // 点击 开始游戏。
Mine.prototype.play = function(event,obj){
    var This = this;
    if(event.which == 1 && obj.className != 'flag'){ //左键
        let curSquare = this.squares[obj.pos[0]][obj.pos[1]];
        let colorName =['zero','one','two','three','four','five','six','seven','eight'];

        if(curSquare.type == 'number'){
            obj.innerHTML = curSquare.value;
            obj.classList.add(colorName[curSquare.value]); // !!! 
            if(curSquare.value == 0){
                // 点到0（0不显示），周围的0都会出现
                obj.innerHTML = '';
                function getAllZero(sq){
                    var aroundZero = This.getAround(sq);
                    for(let i=0;i<aroundZero.length;i++){
                        let x = aroundZero[i][0];
                        let y = aroundZero[i][1];
                        This.tds[x][y].classList.add(colorName[This.squares[x][y].value]);
                        // 递归
                        if(This.squares[x][y].value == 0){
                            if(!This.squares[x][y].check){
                                This.squares[x][y].check =true; // !!! 
                                getAllZero(This.squares[x][y]);
                            }
                        }else{
                            This.tds[x][y].innerHTML = This.squares[x][y].value;

                        }
                    }

                }
                getAllZero(curSquare);
            }
        }else{
            // console.log('你点到雷了!');
            this.gameOver(obj);
        }
        
    }
    if(event.which == 3){ // 右键
        if(obj.className && obj.className != 'flag'){
            return;
        }
        obj.className =obj.className=='flag'?'':'flag';
        if(this.squares[obj.pos[0]][obj.pos[1]].type == 'mine'){
            this.allRight = true;
            
        }else{
            this.allRight = false;
        }
        if(obj.className == 'flag'){
            this.mineNumDom.innerHTML = --this.surplusMine;
        }else{
            this.mineNumDom.innerHTML = ++this.surplusMine;
        }
        if(this.surplusMine == 0){
            if(this.allRight = true){
                alert('游戏通过！');
            }else{
                alert('游戏失败!');
                this.gameOver();
            }
        }
    }
}

      # 按钮切换模式

      • 扫雷游戏有三个模式，通过上方按钮实现模式切换;

var mine = null;
var ln = 0;
var arr = [[9,9,10],[16,16,40],[28,28,99]];
for(let i=0;i<3;i++){
    btns[i].onclick = function(){  // 点击触发事件，切换模式
        btns[ln].className = '';
        this.className = 'active';
        mine = new Mine(...arr[i]);
        mine.init(); // 入口

        if(obj.className && obj.className != 'flag'){
            return;
        }
        obj.className =obj.className=='flag'?'':'flag';
        if(this.squares[obj.pos[0]][obj.pos[1]].type == 'mine'){
            this.allRight = true;
            
        }else{
            this.allRight = false;
        }
        if(obj.className == 'flag'){
            this.mineNumDom.innerHTML = --this.surplusMine;
        }else{
            this.mineNumDom.innerHTML = ++this.surplusMine;
        }
        if(this.surplusMine == 0){
            if(this.allRight = true){
                alert('游戏通过！');
            }else{
                alert('游戏失败!');
                this.gameOver();
            }
        }
    }
}
Mine.prototype.gameOver = function(clickTd){
    // 显示所有的雷；取消点击事件；标红
    for(let i=0;i<this.tr;i++){
        for(let j=0;j<this.td;j++){
            if(this.squares[i][j].type == 'mine'){
                this.tds[i][j].classList.add('mine');
            }
            this.tds[i][j].onmousedown = null;
        }
    }
    if(clickTd){
        clickTd.style.backgroundColor = 'red';
    }
    
}
